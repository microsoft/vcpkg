# Makefile.am: if WINDOWS → strptime.c (POSIX strptime not available on Windows)
if(WIN32)
    list(APPEND SOURCES strptime.c)
endif()

add_library(zvbi ${SOURCES})

target_compile_definitions(zvbi PRIVATE
    HAVE_CONFIG_H
    _CRT_SECURE_NO_WARNINGS
    _CRT_NONSTDC_NO_WARNINGS
    _REENTRANT
    _USE_MATH_DEFINES
)

# lang.c and misc.c contain Latin-1 (ISO 8859-1) characters in string literals.
# vcpkg injects /utf-8 globally (via VCPKG_SET_CHARSET_FLAG); remove it so we
# can specify /source-charset:.1252 for files with non-UTF-8 bytes.
if(MSVC)
    string(REPLACE "/utf-8" "" CMAKE_C_FLAGS "${CMAKE_C_FLAGS}")
    set_source_files_properties(lang.c misc.c PROPERTIES COMPILE_OPTIONS /source-charset:.1252)
    # Promote implicit function declarations to errors — no more C89 surprises
    target_compile_options(zvbi PRIVATE /we4013)
endif()

# The source directory contains config.h, site_def.h, and the sys/time.h compat shim (all generated by the portfile).
# Adding it first ensures #include <sys/time.h> picks up the shim before any system search paths.
target_include_directories(zvbi
    PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
        $<INSTALL_INTERFACE:include>
)

find_package(Iconv REQUIRED)
target_link_libraries(zvbi PRIVATE Iconv::Iconv)

if(WIN32)
    find_package(PThreads4W REQUIRED)
    target_link_libraries(zvbi PRIVATE PThreads4W::PThreads4W ws2_32)
else()
    find_package(Threads REQUIRED)
    target_link_libraries(zvbi PRIVATE Threads::Threads m)
endif()

install(TARGETS zvbi
    EXPORT zvbi-targets
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
)

install(FILES libzvbi.h DESTINATION include)

# ── pkg-config file (FFmpeg's configure uses this to detect zvbi) ──────────────
# Use a placeholder that vcpkg_fixup_pkgconfig() will correct to the installed prefix.
set(_pc_prefix "${CMAKE_INSTALL_PREFIX}")

# List all dependencies in Libs: (not Libs.private:) because FFmpeg's configure
# calls pkg-config --libs without --static, so Libs.private is never returned.
if(WIN32)
    # PThreads4W uses a "d" suffix for debug libs (pthreadVC3d.lib).
    # Query the imported target for the actual library path matching the build type.
    string(TOUPPER "${CMAKE_BUILD_TYPE}" _uc_bt)
    foreach(_prop IMPORTED_IMPLIB_${_uc_bt} IMPORTED_LOCATION_${_uc_bt} IMPORTED_IMPLIB IMPORTED_LOCATION)
        get_target_property(_pthread_implib PThreads4W::PThreads4W ${_prop})
        if(_pthread_implib)
            break()
        endif()
    endforeach()
    if(_pthread_implib)
        get_filename_component(_pthread_libname "${_pthread_implib}" NAME_WE)
    else()
        set(_pthread_libname "pthreadVC3")
    endif()
    set(_pc_extra_libs "-liconv -l${_pthread_libname} -lws2_32")
else()
    set(_pc_extra_libs "-liconv -lpthread -lm")
endif()

file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/zvbi-0.2.pc"
"prefix=${_pc_prefix}
exec_prefix=\${prefix}
libdir=\${prefix}/lib
includedir=\${prefix}/include

Name: ZVBI
Description: VBI Capturing and Decoding Library
Version: 0.2.44
Libs: -L\${libdir} -lzvbi ${_pc_extra_libs}
Cflags: -I\${includedir}
")

install(FILES "${CMAKE_CURRENT_BINARY_DIR}/zvbi-0.2.pc"
    DESTINATION "lib/pkgconfig")
